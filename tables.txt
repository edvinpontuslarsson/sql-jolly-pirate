CREATE TABLE `Boats` (
  `boat_id` INT(8) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `member_id` INT(8) NOT NULL,
  `boat_type` VARCHAR(100) NOT NULL,
  `boat_length` INT(8) NOT NULL,
  KEY `FK` (`member_id`)
);

CREATE TABLE `Berths` (
  `berth_id` INT(8) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `boat_id` INT(8) NOT NULL,
  KEY `FK` (`boat_id`)
);

CREATE TABLE `Events` (
  `event_id` INT(8) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `event_date_utc` VARCHAR(100) NOT NULL,
  `event_info` VARCHAR(5000)
);

CREATE TABLE `Members` (
  `member_id` INT(8) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `member_name` VARCHAR(100) NOT NULL,
  `member_since_utc` VARCHAR(100)
);
